61void DMA1_Initialize(void)
62{
63 //DMA Instance Selection : 0x00
64 DMASELECT = 0x00;
65 //Source Address : lcd_dma_buf
66 DMAnSSA = (volatile unsigned short) &lcd_dma_buf;
67 //Destination Address : &SPI1TXB
68 DMAnDSA = (volatile unsigned short) &SPI1TXB;
69 //DMODE unchanged; DSTP not cleared; SMR GPR; SMODE incremented; SSTP not cleared;
70 DMAnCON1 = 0x02;
71 //Source Message Size : 1
72 DMAnSSZ = 1;
73 //Destination Message Size : 1
74 DMAnDSZ = 1;
75 //Start Trigger : SIRQ SPI1TX;
76 DMAnSIRQ = 0x19;
77 //Abort Trigger : AIRQ None;
78 DMAnAIRQ = 0x00;
79
80 // Clear Destination Count Interrupt Flag bit
81 PIR2bits.DMA1DCNTIF = 0;
82 // Clear Source Count Interrupt Flag bit
83 PIR2bits.DMA1SCNTIF = 0;
84 // Clear Abort Interrupt Flag bit
85 PIR2bits.DMA1AIF = 0;
86 // Clear Overrun Interrupt Flag bit
87 PIR2bits.DMA1ORIF =0;
88
89 PIE2bits.DMA1DCNTIE = 0;
90 PIE2bits.DMA1SCNTIE = 1;
91 DMA1_SetSCNTIInterruptHandler(DMA1_DefaultInterruptHandler);
92 PIE2bits.DMA1AIE = 0;
93 PIE2bits.DMA1ORIE =1;
94 DMA1_SetORIInterruptHandler(DMA1_DefaultInterruptHandler);
95
96 //EN enabled; SIRQEN enabled; DGO not in progress; AIRQEN disabled;
97 DMAnCON0 = 0xC0;
98
99}
