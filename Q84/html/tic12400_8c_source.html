236/*
237 * send tic12400 commands to SPI port 1 with possible delay after transfer
238 * returns 32-bit spi response from the tic12400
239 */
240uint32_t tic12400_wr(const ticbuf_type * buffer, uint16_t del)
241{
242 send_spi1_tic12400_dma((void*) buffer, 4);
243 ticvalue = (uint32_t) spi_link.rxbuf[0]&0x000000ff;
244 ticvalue += (uint32_t) (spi_link.rxbuf[1] << 8) & 0x0000ff00;
245 ticvalue += (uint32_t) ((uint32_t) spi_link.rxbuf[2] << (uint32_t) 16)&0x00ff0000;
246 ticvalue += (uint32_t) ((uint32_t) spi_link.rxbuf[3] << (uint32_t) 24)&0xff000000;
247 tic12400_read_error = false;
248
249 if (ticvalue & parity_fail_v) { // check for command parity errors
250 tic12400_parity_status = true;
251 tic12400_read_error = true;
252 tic12400_parity_count++;
253 MLED_SetHigh();
254 send_spi1_tic12400_dma((void*) &ticstat02, 4);
255 };
256 if (ticvalue & spi_fail_bit_v) {
257 tic12400_read_error = true;
258 tic12400_fail_count++;
259 MLED_SetHigh();
260 send_spi1_tic12400_dma((void*) &ticstat02, 4);
261 }
262 if (del) {
263 WaitMs(1);
264 }
265 return(uint32_t) ticvalue; // raw 32-bit switch register value from the device SO pins
266}

291/*
292 * 32-bit 1's parity check
293 * https://graphics.stanford.edu/~seander/bithacks.html#ParityNaive
294 */
295bool tic12400_parity(uint32_t v)
296{
297 v ^= v >> 16;
298 v ^= v >> 8;
299 v ^= v >> 4;
300 v &= 0xf;
301 return(0x6996 >> v) & 1;
302}
