216 tic12400_init_ok = false;
217 MLED_SetHigh();
218 } else {
219 tic12400_init_ok = true;
220 }
221

225/*
226 * send tic12400 commands to SPI port 1 with possible delay after transfer
227 * returns 32-bit spi response from the tic12400
228 * spi_link.rxbuf[0] is MSB (byte) in the 32-bit value
229 */
230uint32_t tic12400_wr(const ticbuf_type * buffer, uint16_t del)
231{
232 send_spi1_tic12400_dma((void*) buffer, 4);
233 ticvalue = (uint32_t) spi_link.rxbuf[3]&0x000000ff;
234 INTERRUPT_GlobalInterruptHighDisable(); // buffer input bits
235 in_buf1 = spi_link.rxbuf[3];
236 in_buf2 = spi_link.rxbuf[2];
237 in_buf3 = spi_link.rxbuf[1];
238 INTERRUPT_GlobalInterruptHighEnable();
239 ticvalue += (uint32_t) (spi_link.rxbuf[2] << 8) & 0x0000ff00;
240 ticvalue += (uint32_t) ((uint32_t) spi_link.rxbuf[1] << (uint32_t) 16)&0x00ff0000;
241 ticvalue += (uint32_t) ((uint32_t) spi_link.rxbuf[0] << (uint32_t) 24)&0xff000000;
242 tic12400_read_error = false;
243
244 if (spi_link.rxbuf[0] & parity_fail_v) { // check for command parity errors
245 tic12400_parity_status = true;
246 tic12400_read_error = true;
247 tic12400_parity_count++;
248 send_spi1_tic12400_dma((void*) &ticstat02, 4);
249 };
250 if (spi_link.rxbuf[0] & spi_fail_bit_v) {
251 tic12400_read_error = true;
252 tic12400_fail_count++;
253 send_spi1_tic12400_dma((void*) &ticstat02, 4);
254 }
255 if (del) {
256 WaitMs(del);
257 }
258 return(uint32_t) ticvalue; // raw 32-bit switch register value from the device SO pins
259}

281/*
282 * 32-bit 1's parity check
283 * https://graphics.stanford.edu/~seander/bithacks.html#ParityNaive
284 */
285bool tic12400_parity(uint32_t v)
286{
287 v ^= v >> 16;
288 v ^= v >> 8;
289 v ^= v >> 4;
290 v &= 0xf;
291 return(0x6996 >> v) & 1;
292}
