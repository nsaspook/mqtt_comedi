53void init_slaveo(void)
54{
55 uint8_t val = 0xff;
56
57 SPI2CON0bits.EN = 0;
58 SPI2_SetRxInterruptHandler(slaveo_rx_isr);
59 SPI2_SetInterruptHandler(slaveo_spi_isr);
60 while (!SPI2STATUSbits.RXRE) { // clear the FIFO of data
61 val = SPI2RXB;
62 }
63 serial_buffer_ss.data[2] = val;
64 SPI2STATUSbits.RXRE = 0;
65 SPI2CON2bits.TXR = 1; // FIFO required for transmit
66 SPI2CON0bits.EN = 1;
67 ADC_SelectContext(CONTEXT_1);
68}

70void slaveo_rx_isr(void)
71{
72 uint8_t command;
73
74 /* we only get this when the master wants data, the slave never generates one */
75 // SPI port #2 SLAVE receiver
76
77 TP1_SetHigh();
78 DLED_SetHigh();
79#ifdef SLAVE_DEBUG
80 if (SPI2INTFbits.RXOIF) {
81 spi_stat_ss.rxof_bit++;
82 }
83
84 if (SPI2STATUSbits.SPI2RXRE) {
85 } else {
86 if (spi_comm_ss.PORT_DATA) {
87 spi_stat_ss.spi_noerror_count++;
88 }
89 }
90#endif
91
92 data_in2 = SPI2RXB;
93 serial_buffer_ss.data[serial_buffer_ss.raw_index] = data_in2;
94
95 // PORT_GO_BYTES
96 if (serial_buffer_ss.make_value) {
97 if (serial_buffer_ss.raw_index == PORT_GO_BYTES) {
98 V.bmc_do = serial_buffer_ss.data[1];
99 V.bmc_do += (uint32_t) (serial_buffer_ss.data[2] << 8u)&0x0000ff00;
100 data_in2 = 0;
101 serial_buffer_ss.make_value = false;
102 serial_buffer_ss.raw_index = 0;
103 spi_stat_ss.txdone_bit++; // number of completed packets
104 spi_stat_ss.slave_tx_count++;
105 data_in2 = 0;
106 } else {
107 spi_stat_ss.slave_tx_count++;
108 data_in2 = 0;
109 }
110 }
111
112 // DAC_GO_BYTES
113 if (serial_buffer_ss.dac_value) {
114 if (serial_buffer_ss.raw_index == DAC_GO_BYTES) {
115 V.bmc_ao = serial_buffer_ss.data[1];
116 serial_buffer_ss.dac_value = false;
117 serial_buffer_ss.raw_index = 0;
118 spi_stat_ss.txdone_bit++; // number of completed packets
119 spi_stat_ss.slave_tx_count++;
120 data_in2 = 0;
121 } else {
122 spi_stat_ss.slave_tx_count++;
123 data_in2 = 0;
124 }
125 }
126
127 // PORT_GET_BYTES
128 if (serial_buffer_ss.get_value) {
129 if (serial_buffer_ss.raw_index == PORT_GET_BYTES) {
130 SPI2TXB = (uint8_t) V.bmc_di >> ((uint8_t) 8 * (uint8_t) (serial_buffer_ss.raw_index));
131 serial_buffer_ss.get_value = false;
132 serial_buffer_ss.raw_index = 0;
133 spi_stat_ss.txdone_bit++; // number of completed packets
134 data_in2 = 0;
135 } else {
136 spi_stat_ss.slave_tx_count++;
137 SPI2TXB = (uint8_t) V.bmc_di >> ((uint8_t) 8 * (uint8_t) (serial_buffer_ss.raw_index));
138 data_in2 = 0;
139 }
140 }
141
142 // ADC_GET_BYTES
143 if (serial_buffer_ss.adc_value) {
144 if (serial_buffer_ss.raw_index == ADC_GET_BYTES) {
145 SPI2TXB = ((adc_buffer[channel] >> 8)&0x00ff);
146 serial_buffer_ss.adc_value = false;
147 serial_buffer_ss.raw_index = 0;
148 spi_stat_ss.txdone_bit++; // number of completed packets
149 data_in2 = 0;
150 } else {
151 spi_stat_ss.slave_tx_count++;
152 if (serial_buffer_ss.raw_index == 1) {
153 SPI2TXB = (adc_buffer[channel] &0x00ff);
154 } else {
155 SPI2TXB = ((adc_buffer[channel] >> 8)&0x00ff);
156 }
157 data_in2 = 0;
158 }
159 }
160
161 // GET_CFG_BYTES
162 if (serial_buffer_ss.cfg_value) {
163 if (serial_buffer_ss.raw_index == CFG_GET_BYTES) {
164 SPI2TXB = spi_stat_ss.daq_conf; // respond with DAQ configuration bits for unregistered inputs
165 serial_buffer_ss.cfg_value = false;
166 serial_buffer_ss.raw_index = 0;
167 spi_stat_ss.txdone_bit++; // number of completed packets
168 data_in2 = 0;
169 } else {
170 spi_stat_ss.slave_tx_count++;
171 if (serial_buffer_ss.raw_index == 1) {
172 SPI2TXB = serial_buffer_ss.raw_index;
173 } else {
174 SPI2TXB = serial_buffer_ss.raw_index;
175 }
176 data_in2 = 0;
177 }
178 }
179
180 if (++serial_buffer_ss.raw_index > SPI_BUFFER_LEN - 1) {
181 serial_buffer_ss.raw_index = 0;
182 spi_stat_ss.txuf_bit++; // buffer high watermark cleared
183 }
184
185 command = data_in2 & HI_NIBBLE;
186
187 if (serial_buffer_ss.dac_value || serial_buffer_ss.get_value || serial_buffer_ss.make_value || serial_buffer_ss.dac_value || serial_buffer_ss.cfg_value) {
188 goto isr_end;
189 }
190
191 if (command == CMD_DAC_GO) { // Found a GO for a DAC conversion command
192 spi_comm_ss.ADC_RUN = false;
193 spi_comm_ss.PORT_DATA = false;
194 spi_stat_ss.dac_count++;
195 serial_buffer_ss.raw_index = 1;
196 serial_buffer_ss.dac_value = true;
197 spi_stat_ss.slave_tx_count++;
198 spi_comm_ss.REMOTE_LINK = true;
199 TMR0_Reload();
200 }
201
202 if (command == CMD_ADC_GO) { // Found a GO for a conversion command
203 spi_comm_ss.ADC_RUN = false;
204 spi_comm_ss.PORT_DATA = false;
205 spi_stat_ss.adc_count++;
206 channel = data_in2 & LO_NIBBLE; // only 16 possible channels so higher numbers needs to be munged
207 if (channel > AI_CHAN_FIX) {
208 switch (channel) {
209 case 0x6:
210 channel = channel_ANC6;
211 break;
212 case 0x7:
213 channel = channel_ANC7;
214 break;
215 case 0x8:
216 channel = channel_AND5;
217 break;
218 case 0x9:
219 channel = channel_VSS;
220 break;
221 case 0xa:
222 channel = channel_Temp;
223 break;
224 case 0xb:
225 channel = channel_DAC1;
226 break;
227 case 0xc:
228 channel = channel_FVR_Buffer1;
229 break;
230 case 0xd:
231 channel = channel_FVR_Buffer2;
232 break;
233 case 0xf:
234 SPI1CON0bits.EN = 0;
235 MLED_SetHigh();
236 serial_buffer_ss.raw_index = 0;
237 serial_buffer_ss.adc_value = false;
238 serial_buffer_ss.make_value = false;
239 serial_buffer_ss.dac_value = false;
240 serial_buffer_ss.get_value = false;
241 serial_buffer_ss.cfg_value = false;
242 MLED_SetLow();
243 SPI1CON0bits.EN = 1;
244 channel = channel_ANA0;
245 break;
246 default:
247 channel = channel_ANA0;
248 break;
249 }
250 }
251 serial_buffer_ss.raw_index = 1;
252 serial_buffer_ss.adc_value = true;
253 spi_comm_ss.REMOTE_LINK = true;
254 TMR0_Reload();
255 }
256
257 if (!V.di_fail && command == CMD_PORT_GET) { // send the V.bmc_di buffer
258 spi_comm_ss.ADC_RUN = false;
259 spi_comm_ss.PORT_DATA = true;
260 spi_stat_ss.port_count++;
261 serial_buffer_ss.raw_index = 0;
262 serial_buffer_ss.get_value = true;
263 spi_comm_ss.REMOTE_LINK = true;
264 TMR0_Reload();
265 }
266
267 if (!V.do_fail && command == CMD_PORT_GO) { // Found a GO for a DO command
268 spi_comm_ss.ADC_RUN = false;
269 spi_comm_ss.PORT_DATA = true;
270 spi_stat_ss.port_count++;
271 channel = data_in2;
272 serial_buffer_ss.raw_index = 1;
273 serial_buffer_ss.make_value = true;
274 spi_stat_ss.slave_tx_count++;
275 spi_comm_ss.REMOTE_LINK = true;
276 TMR0_Reload();
277 }
278
279 if (command == CMD_DUMMY_CFG) {
280 serial_buffer_ss.raw_index = 1;
281 serial_buffer_ss.cfg_value = false;
282 if (!serial_buffer_ss.adc_value && !serial_buffer_ss.dac_value && !serial_buffer_ss.make_value && !serial_buffer_ss.get_value) {
283 MLED_SetHigh();
284 serial_buffer_ss.adc_value = false;
285 serial_buffer_ss.make_value = false;
286 serial_buffer_ss.dac_value = false;
287 serial_buffer_ss.get_value = false;
288 MCZ_PWM_SetLow();
289 MLED_SetLow();
290 }
291 TMR0_Reload();
292 StartTimer(TMR_ADC, ADCDELAY);
293 }
294
295isr_end:
296 spi_stat_ss.slave_int_count++;
297 DLED_SetLow();
298 TP1_SetLow();
299}
