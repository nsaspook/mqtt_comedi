58void DSM_Initialize(void)
59{
60 // set the DSM to the options selected in the User Interface
61
62 // Modulator Output Pin Polarity is not inverted
63 MD1CON0bits.MD1OPOL = 0;
64
65 // Modulation Low Carrier Synchronization is enabled
66 MD1CON1bits.MD1CLSYNC = 1;
67
68 // Modulation Low Carrier Polarity is not inverted
69 MD1CON1bits.MD1CLPOL = 0;
70
71 // Modulation High Carrier Synchronization is enabled
72 MD1CON1bits.MD1CHSYNC = 1;
73
74 // Modulation High Carrier Polarity is not inverted
75 MD1CON1bits.MD1CHPOL = 0;
76
77 // MD1MS UART1_TX;
78 MD1SRC = 0x13;
79
80 // MD1CH NCO1OUT;
81 MD1CARH = 0x0E;
82
83 // MD1CL CCP1;
84 MD1CARL = 0x04;
85
86 // Enable the Modulator
87 MD1CON0bits.MD1EN = 1;
88}

90void DSM_ManualModulationSet (void)
