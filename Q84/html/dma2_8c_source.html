62void DMA2_Initialize(void)
63{
64 //DMA Instance Selection : 0x01
65 DMASELECT = 0x01;
66 //Source Address : &SPI1RXB
67 DMAnSSA = (volatile unsigned short)&SPI1RXB;
68 //Destination Address : &spi1_rec_buf
69 DMAnDSA= (volatile unsigned short)&spi1_rec_buf;
70 //DMODE incremented; DSTP not cleared; SMR SFR; SMODE unchanged; SSTP not cleared;
71 DMAnCON1 = 0x40;
72 //Source Message Size : 1
73 DMAnSSZ = 1;
74 //Destination Message Size : 4
75 DMAnDSZ = 8;
76 //Start Trigger : SIRQ None;
77 DMAnSIRQ = 0x00;
78 //Abort Trigger : AIRQ None;
79 DMAnAIRQ = 0x00;
80
81 // Clear Destination Count Interrupt Flag bit
82 PIR6bits.DMA2DCNTIF = 0;
83 // Clear Source Count Interrupt Flag bit
84 PIR6bits.DMA2SCNTIF = 0;
85 // Clear Abort Interrupt Flag bit
86 PIR6bits.DMA2AIF = 0;
87 // Clear Overrun Interrupt Flag bit
88 PIR6bits.DMA2ORIF =0;
89
90 PIE6bits.DMA2DCNTIE = 1;
91 DMA2_SetDCNTIInterruptHandler(DMA2_DefaultInterruptHandler);
92 PIE6bits.DMA2SCNTIE = 1;
93 DMA2_SetSCNTIInterruptHandler(DMA2_DefaultInterruptHandler);
94 PIE6bits.DMA2AIE = 0;
95 PIE6bits.DMA2ORIE =1;
96 DMA2_SetORIInterruptHandler(DMA2_DefaultInterruptHandler);
97
98 //EN enabled; SIRQEN disabled; DGO not in progress; AIRQEN disabled;
99 DMAnCON0 = 0x80;
100
101}
