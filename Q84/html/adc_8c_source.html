65void ADC_Initialize(void)
66{
67 //ADACT disabled;
68 ADACT = 0x00;
69
70 //ADCCS FOSC/48;
71 ADCLK = 0x17;
72
73 //ADC charge pump control
74 ADCP = 0x00;
75
76
77 /****************************************
78 * Configure Context-1 *
79 ****************************************/
80 ADCTX = 0x0;
81
82 //ADLTHL 0;
83 ADLTHL = 0x00;
84
85 //ADLTHH 0;
86 ADLTHH = 0x00;
87
88 //ADUTHL 0;
89 ADUTHL = 0x00;
90
91 //ADUTHH 0;
92 ADUTHH = 0x00;
93
94 //ADSTPTL 0;
95 ADSTPTL = 0x00;
96
97 //ADSTPTH 0;
98 ADSTPTH = 0x00;
99
100 //ADACCL 0;
101 ADACCL = 0x00;
102
103 //ADACCH 0;
104 ADACCH = 0x00;
105
106 //ADACCU 0;
107 ADACCU = 0x00;
108
109 //ADCNT 0;
110 ADCNT = 0x00;
111
112 //ADRPT 0;
113 ADRPT = 0x00;
114
115 //ADRESL 0;
116 ADRESL = 0x00;
117
118 //ADRESH 0;
119 ADRESH = 0x00;
120
121 //ADCHS ANA0;
122 ADPCH = 0x00;
123
124 //ADACQL 20;
125 ADACQL = 0x14;
126
127 //ADACQH 0;
128 ADACQH = 0x00;
129
130 //ADCAP Additional uC of 31pF;
131 ADCAP = 0x1F;
132
133 //ADPREL 0;
134 ADPREL = 0x00;
135
136 //ADPREH 0;
137 ADPREH = 0x00;
138
139 //ADCONT disabled;
140 ADCON0 = 0x00;
141
142 //ADPPOL Vss; ADIPEN disabled; ADGPOL digital_low; ADDSEN enabled;
143 ADCON1 = 0x01;
144
145 //ADPSIS RES; ADCRS 1; ADACLR disabled; ADMD Basic_mode;
146 ADCON2 = 0x10;
147
148 //ADCALC First derivative of Single measurement; ADSOI ADGO not cleared; ADTMD disabled;
149 ADCON3 = 0x00;
150
151 //ADMATH registers not updated;
152 ADSTAT = 0x00;
153
154 //ADNREF VSS; ADPREF external;
155 ADREF = 0x02;
156
157 //CHEN channel content is not included; SSI scan sequence continues;
158 ADCSEL1 = 0x00;
159
160 //Clear ADC Interrupt Flag
161 PIR1bits.ADIF = 0;
162
163 //Clear ADC active clock tuning interrupt flag
164 PIR1bits.ACTIF = 0;
165
166 //Clear ADC Context Threshold Interrupt Flag
167 PIR2bits.ADCH1IF = 0;
168
169
170 //Configure interrupt handlers
171 ADC_SetADIInterruptHandler(ADC_DefaultADI_ISR);
172 ADC_SetContext1ThresholdInterruptHandler(ADC_DefaultContext1Threshold_ISR);
173 ADC_SetActiveClockTuningInterruptHandler(ADC_DefaultActiveClockTuning_ISR);
174
175 //ADON enabled; CSEN disabled; ADCS FOSC/ADCLK; ADFM right; GO_nDONE stop;
176 ADCON0 = 0x84;
177}

223void ADC_StartConversion(ADC_channel_t channel)
224{
225 //Select the A/D channel
226 ADPCH = channel;
227
228 //Turn ON the ADC module
229 ADCON0bits.ON = 1;
230
231 //Start the conversion
232 ADCON0bits.GO = 1;
233}

246adc_result_t ADC_GetSingleConversion(ADC_channel_t channel)
247{
248 //Select the A/D channel
249 ADPCH = channel;
250
251 //Turn ON the ADC module
252 ADCON0bits.ON = 1;
253
254 //Disable the continuous mode
255 ADCON0bits.CONT = 0;
256
257 //Start the conversion
258 ADCON0bits.GO = 1;
259
260 //Wait for the conversion to finish
261 while(ADCON0bits.GO)
262 {
263
264 }
265
266 return ((adc_result_t) ((ADRESH << 8) + ADRESL));
267}
