115/*
116 * add the required CRC bytes to a MODBUS message
117 */
118static uint16_t modbus_rtu_send_msg_crc(volatile uint8_t *req, uint16_t req_length)
119{
120 uint16_t crc;
121
122 crc = crc16(req, req_length);
123 req[req_length++] = crc >> (uint16_t) 8;
124 req[req_length++] = crc & 0x00FF;
125
126 return req_length;
127}

129/*
130 * constructs a properly formatted RTU message with CRC from a program memory array to the data memory array buffer
131 */
132uint16_t modbus_rtu_send_msg(void *cc_buffer, const void *modbus_cc_mode, uint16_t req_length)
133{
134 memcpy((void*) cc_buffer, (const void *) modbus_cc_mode, req_length);

136 * add the CRC and increase message size by two bytes for the CRC16
137 */
138 return modbus_rtu_send_msg_crc((volatile uint8_t *) cc_buffer, req_length);

141/*
142 * calculate a CRC16 from the data buffer
143 */
144uint16_t crc16(volatile uint8_t *buffer, uint16_t buffer_length)
145{
146 uint8_t crc_hi = 0xFF; /* high CRC byte initialized */
147 uint8_t crc_lo = 0xFF; /* low CRC byte initialized */
148 uint8_t i; /* will index into CRC lookup */
149 uint16_t crc16t;
150
151 /* pass through message buffer */
152 while (buffer_length--) {
153 i = crc_hi ^ *buffer++; /* calculate the CRC */
154 crc_hi = crc_lo ^ table_crc_hi[i];
155 crc_lo = table_crc_lo[i];
156 }
157
158 crc16t = (uint16_t) crc_hi << (uint16_t) 8 | (uint16_t) crc_lo;
159 return crc16t;
160}

221/*
222 * reorder 16-bit word bytes for int32_t
223 * https://control.com/forums/threads/endianness-for-32-bit-data.48584/
224 * https://ctlsys.com/support/common_modbus_protocol_misconceptions/
225 * https://iotech.force.com/edgexpert/s/article/Byte-and-Word-Swapping-in-Modbus
226 *
227 * "Little Endian" slaves or "Big Endian" slaves
228 * Byte endianness with Word endianness?
229 * Lions and Tigers and Bears!
230 */
231int32_t mb32_swap(const int32_t value)
232{
233 uint8_t i;
234 union MREG32 dvalue;
235
236 // program it simple and easy to understand way, let the compiler optimize the expressions
237 dvalue.value = value;
238 i = dvalue.bytes[0];
239 dvalue.bytes[0] = dvalue.bytes[1];
240 dvalue.bytes[1] = i;
241 i = dvalue.bytes[2];
242 dvalue.bytes[2] = dvalue.bytes[3];
243 dvalue.bytes[3] = i;
244 return dvalue.value;
245}

260/*
261 * Simple MODBUS master state machine
262 * this needs to run in the main programming loop
263 * to handle RS485 serial I/O exchanges
264 */
265int8_t master_controller_work(C_data * client)
266{
267 static uint32_t spacing = 0;
268
269 if (spacing++ <SPACING && !M.rx) {
270 return T_spacing;
271 }
272
273 spacing = 0;
274
275 client->trace = T_begin;
276 switch (client->cstate) {
277 case CLEAR:
278 client->trace = T_clear;
279 clear_2hz();
280 clear_500ahz();
281 client->cstate = INIT;
282 client->modbus_command = client->mcmd++; // sequence modbus commands to client
283 if (client->modbus_command == G_CONFIG && client->config_ok) { // skip if we have valid data from client
284 client->modbus_command = client->mcmd++;
285 }
286 if (client->modbus_command == G_PASSWD && client->passwd_ok) { // skip if we have valid data from client
287 client->modbus_command = client->mcmd++;
288 }
289 if (client->modbus_command == G_LIGHT && client->light_ok) { // skip if we have valid data from client
290 client->modbus_command = client->mcmd++;
291 }
292 if (client->modbus_command == G_VERSION && client->version_ok) { // skip if we have valid data from client
293 client->modbus_command = client->mcmd++;
294 }
295 if (client->modbus_command == G_SERIAL && client->serial_ok) { // skip if we have valid data from client
296 client->modbus_command = client->mcmd++;
297 }
298 if (client->mcmd > G_LAST) {
299 client->mcmd = G_ID;
300 }
301 /*
302 * command specific tx buffer setup
303 */
304 switch (client->modbus_command) {
305 case G_VERSION: // write code request
306 client->trace = T_version;
307 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_version, sizeof(modbus_em_version));
308 break;
309 case G_SERIAL: // write code request
310 client->trace = T_serial;
311 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_serial, sizeof(modbus_em_serial));
312 break;
313 case G_LIGHT: // write code request
314 client->trace = T_light;
315 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_light, sizeof(modbus_em_light));
316 break;
317 case G_PASSWD: // write code request
318 client->trace = T_passwd;
319 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_passwd, sizeof(modbus_em_passwd));
320 break;
321 case G_CONFIG: // write code request
322 client->trace = T_config;
323 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_config, sizeof(modbus_em_config));
324 break;
325 case G_DATA1: // read code request
326 client->trace = T_data;
327 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_data1, sizeof(modbus_em_data1));
328 break;
329 case G_DATA2: // read code request
330 client->trace = T_data;
331 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_data2, sizeof(modbus_em_data2));
332 break;
333 case G_LAST: // end of command sequences
334 client->cstate = CLEAR;
335 client->mcmd = G_ID; // what do we run next
336 break;
337 case G_ID: // operating mode request
338 client->trace = T_id;
339 default:
340 client->req_length = modbus_rtu_send_msg((void*) cc_buffer_tx, (const void *) modbus_em_id, sizeof(modbus_em_id));
341 break;
342 }
343 break;
344 case INIT:
345 client->trace = T_init;
346 /*
347 * MODBUS master query speed
348 */
349#ifdef FASTQ
350 if (get_500ahz(false) >= CDELAY) {
351#else
352 if (get_2hz(false) >= QDELAY) {
353#endif
354#ifndef AUTO_DERE
355 half_dup_tx(false); // no delays here
356#endif
357 M.recv_count = 0;
358 client->cstate = SEND;
359 clear_500hz();
360 client->trace = T_init_d;
361 }
362 break;
363 case SEND:
364 client->trace = T_send;
365 if (get_500hz(false) >= TEDELAY) {
366 for (uint8_t i = 0; i < client->req_length; i++) {
367 Swrite(cc_buffer_tx[i]);
368 }
369 client->cstate = RECV;
370 clear_500hz(); // state machine execute background timer clear
371 client->trace = T_send_d;
372 M.sends++;
373 M.rx = false;
374 if (serial_trmt()) { // check for serial UART transmit shift register and buffer empty
375 clear_500hz(); // clear timer until buffer empty
376 }
377#ifndef AUTO_DERE
378 DERE_SetLow(); // enable modbus receiver
379#endif
380 }
381 break;
382 case RECV:
383 client->trace = T_recv;
384 if (get_500hz(false) >= TEDELAY) { // state machine execute timer test
385
386 client->trace = T_recv_r;
387#ifndef AUTO_DERE
388 half_dup_rx(false); // no delays here
389#endif
390
391 /*
392 * check received response data for size and format for each command sent
393 */
394 switch (client->modbus_command) {
395 case G_LIGHT: // check for controller back-light codes
396 modbus_write_check(client, &client->light_ok, sizeof(em_light));
397 break;
398 case G_PASSWD: // check for controller password codes
399 modbus_write_check(client, &client->passwd_ok, sizeof(em_passwd));
400 break;
401 case G_CONFIG: // check for controller configuration codes
402 modbus_write_check(client, &client->config_ok, sizeof(em_config));
403 break;
404 case G_DATA1: // check for controller data1 codes
405 modbus_read_check(client, &client->data_ok, sizeof(em_data1), em_data_handler);
406 break;
407 case G_DATA2: // check for controller data2 codes
408 modbus_read_check(client, &client->data_ok, sizeof(em_data2), emt_data_handler);
409 break;
410 case G_VERSION: // check for controller EM540 firmware codes
411 modbus_read_check(client, &client->version_ok, sizeof(em_version), emv_data_handler);
412 break;
413 case G_SERIAL: // check for controller EM540 serial codes
414 modbus_read_check(client, &client->serial_ok, sizeof(em_serial), ems_data_handler);
415 break;
416 case G_ID: // check for client module type
417 default:
418 modbus_read_id_check(client, &client->id_ok, sizeof(em_id));
419 break;
420 }
421 }
422 break;
423 default:
424 break;
425 }
426 return client->trace;
427}

533/*
534 * check if we are done with interrupt background buffered transmission of serial data with FIFO
535 *
536 * TRMT: Transmit Shift Register is Empty bit (read-only)
537 * 1 = Transmit shift register is empty and transmit buffer is empty (the last transmission has completed)
538 * 0 = Transmit shift register is not empty, a transmission is in progress or queued in the transmit buffer
539 *
540 * ? 8-level deep First-In-First-Out (FIFO) transmit data buffer, ? 8-level deep FIFO receive data buffer
541 * Interrupt is generated and asserted while the transmit buffer is empty
542 *
543 * so this will return 'true' after the buffer is empty 'interrupt' and after the last bit is on the wire
544 */
545
546static bool serial_trmt(void)
547{
548 return !(Strmt); // note, we invert the TRMT bit so it's true while transmitting
549}

592static bool modbus_write_check(C_data * client, bool* cstate, const uint16_t rec_length)
593{
594 uint16_t c_crc, c_crc_rec;
595
596#ifdef MB_EM540
597 client->req_length = rec_length;
598 if (DBUG_R((M.recv_count >= client->req_length) && (cc_buffer[0] == MADDR) && (cc_buffer[1] == WRITE_SINGLE_REGISTER))) {
599 c_crc = crc16(cc_buffer, client->req_length - 2);
600 c_crc_rec = crc16_receive(client);
601 if (DBUG_R c_crc == c_crc_rec) {
602 *cstate = true;
603 MM_ERROR_C;
604 } else {
605 *cstate = false;
606 log_crc_error(c_crc, c_crc_rec);
607 }
608 client->cstate = CLEAR; // where do we go next
609 client->mcmd = G_LAST; // what do we run next
610 } else {
611 if (get_500hz(false) >= RDELAY) {
612 client->cstate = CLEAR; // where do we go next
613 client->mcmd = G_ID; // what do we run next
614 M.to_error++;
615 M.error++;
616 if (client->data_ok) {
617 MM_ERROR_C;
618 }
619 }
620 }
621#endif
622 return *cstate;
623}
