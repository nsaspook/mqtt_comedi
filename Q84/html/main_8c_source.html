1017device_id_data_t DeviceID_Read(device_id_address_t address)
1018{
1019 device_id_data_t deviceID;
1020
1021 //Save the table pointer
1022 uint32_t tablePointer = ((uint32_t) TBLPTRU << 16) | ((uint32_t) TBLPTRH << 8) | ((uint32_t) TBLPTRL);
1023
1024 //Load table pointer with Device ID address
1025 TBLPTRU = (uint8_t) (address >> 16);
1026 TBLPTRH = (uint8_t) (address >> 8);
1027 TBLPTRL = (uint8_t) address;
1028
1029 //Execute table read and increment table pointer
1030 asm("TBLRD*+");
1031
1032 deviceID = (device_id_data_t) TABLAT;
1033
1034 //Execute table read
1035 asm("TBLRD*");
1036
1037 deviceID |= (device_id_data_t) (TABLAT << 8);
1038
1039 //Restore the table pointer
1040 TBLPTRU = (uint8_t) (tablePointer >> 16);
1041 TBLPTRH = (uint8_t) (tablePointer >> 8);
1042 TBLPTRL = (uint8_t) tablePointer;
1043
1044 return deviceID;
1045}
