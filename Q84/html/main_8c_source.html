938device_id_data_t DeviceID_Read(device_id_address_t address)
939{
940 device_id_data_t deviceID;
941
942 //Save the table pointer
943 uint32_t tablePointer = ((uint32_t) TBLPTRU << 16) | ((uint32_t) TBLPTRH << 8) | ((uint32_t) TBLPTRL);
944
945 //Load table pointer with Device ID address
946 TBLPTRU = (uint8_t) (address >> 16);
947 TBLPTRH = (uint8_t) (address >> 8);
948 TBLPTRL = (uint8_t) address;
949
950 //Execute table read and increment table pointer
951 asm("TBLRD*+");
952
953 deviceID = (device_id_data_t) TABLAT;
954
955 //Execute table read
956 asm("TBLRD*");
957
958 deviceID |= (device_id_data_t) (TABLAT << 8);
959
960 //Restore the table pointer
961 TBLPTRU = (uint8_t) (tablePointer >> 16);
962 TBLPTRH = (uint8_t) (tablePointer >> 8);
963 TBLPTRL = (uint8_t) tablePointer;
964
965 return deviceID;
966}
