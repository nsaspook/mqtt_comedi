18/*
19 * send and receive USB data
20 * using the asynchronous hid device mode for this driver
21 */
22int32_t SendUSBCmd(hid_device *handle, uint8_t *cmdBuf, uint8_t *responseBuf)
23{
24 int32_t r;
25
26 r = hid_write(handle, cmdBuf, COMMAND_BUFFER_LENGTH);
27 if (r < 0) {
28 return ERROR_UNABLE_TO_WRITE_TO_DEVICE;
29 }
30
31 //when the hid device is configured as synchronous, the first
32 //hid_read returns the desired results. and the while() loop
33 //is skipped.
34 //
35 //when the hid device is configured as asynchronous, the first
36 //hid_read may or may not succeed, depending on the latency
37 //of the attached device. When no data is returned, r = 0 and
38 //the while loop keeps polling the returned data until it is
39 //received.
40 r = hid_read(handle, responseBuf, RESPONSE_BUFFER_LENGTH);
41 if (r < 0) {
42 return ERROR_UNABLE_TO_READ_FROM_DEVICE;
43 }
44
45 while (r == 0) {
46 r = hid_read(handle, responseBuf, RESPONSE_BUFFER_LENGTH);
47 if (r < 0) {
48 return ERROR_UNABLE_TO_READ_FROM_DEVICE;
49 }
50 sleep_us(SPI_STATUS_DELAY_US);
51 }
52
53 return responseBuf[1];
54}

66 if (!microseconds) {

70 ts.tv_nsec = (microseconds % 1000000) * 1000; // remainder, in nanoseconds

147/*
148 * open the USB device with a optional serial number
149 */
150mcp2210_spi_type* hidrawapi_mcp2210_init(const wchar_t *serial_number)
151{
152 printf("\r\n--- MCP2210 Driver Version %s %s %s ---\r\n", MCP2210_DRIVER, build_date, build_time);
153 cbufs(); // clear command and response buffers
154 //------------------ Open MCP2210 and display info ---------------
155 printf("Open Device: 04d8:00de\n");
156 // Open the device using the VID(vendor ID, PID(product ID),
157 // and optionally the Serial number.
158 S->handle = hid_open(0x4d8, 0xde, serial_number); // open the MCP2210 device
159 if (!S->handle) {
160 printf("unable to open the MCP2210\n");
161 return NULL;
162 }
163 // Read the Manufacturer String
164 wstr[0] = 0x0000;
165 S->res = hid_get_manufacturer_string(S->handle, wstr, MAX_STR);
166 if (S->res < 0) {
167 printf("Unable to read manufacturer string\n");
168 }
169 printf("Manufacturer String: %ls\n", wstr);
170
171 // Read the Product String
172 wstr[0] = 0x0000;
173 S->res = hid_get_product_string(S->handle, wstr, MAX_STR);
174 if (S->res < 0) {
175 printf("Unable to read product string\n");
176 }
177 printf("Product String: %ls\n", wstr);
178
179 // Read the Serial Number String
180 wstr[0] = 0x0000;
181 S->res = hid_get_serial_number_string(S->handle, wstr, MAX_STR);
182 if (S->res < 0) {
183 printf("Unable to read serial number string\n");
184 }
185 printf("Serial Number String: (%d) %ls", wstr[0], wstr);
186 printf("\n");
187
188 hid_set_nonblocking(S->handle, 1); // async operation, don't block
189
190 //-------------- Set GPIO pin function (0x21) -------------
191 // configure chip selects and interrupts for all devices on the SPI buss
192 cbufs();
193 S->buf[0] = 0x21; // command 21 - set GPIO pin's functions
194 S->buf[4] = 0x01; // GPIO 0 set to 0x01 - SPI CS, BMX160
195 S->buf[5] = 0x00; // GPIO 1
196 S->buf[6] = 0x00; // GPIO 2
197 S->buf[7] = 0x02; // GPIO 3, act led
198 S->buf[8] = 0x02; // GPIO 4, LOWPWR led
199 S->buf[9] = 0x01; // GPIO 5 set to 0x01 - SPI CS, tic12400
200 S->buf[10] = 0x02; // GPIO 6 external interrupt input
201 S->buf[11] = 0x01; // GPIO 7 set to 0x01 - SPI CS, MC33996
202 S->buf[12] = 0x00; // GPIO 8
203 S->buf[13] = 0xff;
204 S->buf[14] = 0x01;
205 S->buf[15] = 0b01000000; // set GPIO 6 to input
206 S->buf[16] = 0b00000001; //
207 S->buf[17] = 0b00000010; // count Falling edges
208 S->res = SendUSBCmd(S->handle, S->buf, S->rbuf);
209
210 // ------------ Set GPIO pin direction (0x32)--------------
211 cbufs();
212 S->buf[0] = 0x32; // command 32 - set GPIO pin's directions
213 // function: 0 = output, 1 = input
214 S->buf[4] = 0b01000000; // set GPIO 0-5,7 to outputs, GPIO 6 for input
215 S->buf[5] = 0b00000001; // set GPIO 8 to input
216 S->res = SendUSBCmd(S->handle, S->buf, S->rbuf);
217
218 // ------------ Set GPIO pin level (0x30)--------------
219 cbufs();
220 S->buf[0] = 0x30; // command 30 - set GPIO pin's level to all high
221 S->buf[4] = 0xff;
222 S->buf[5] = 0xff;
223 S->res = SendUSBCmd(S->handle, S->buf, S->rbuf);
224
225 cancel_spi_transfer(); // reset the SPI engine
226 printf("MCP2210 init complete\n"); // ctrl c to exit program
227 return S;
228}
