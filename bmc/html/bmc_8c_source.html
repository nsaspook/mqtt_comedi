97void led_lightshow(int speed)
98{
99 static int j = 0;
100 static uint8_t cylon = 0xff;
101 static int alive_led = 0;
102 static bool LED_UP = true;
103
104 if (j++ >= speed) { // delay a bit ok
105 if (0) { // screen status feedback
106 bmc.dataout.dio_buf = ~cylon; // roll leds cylon style
107 } else {
108 bmc.dataout.dio_buf = cylon; // roll leds cylon style (inverted)
109 }
110
111 if (LED_UP && (alive_led != 0)) {
112 alive_led = alive_led * 2;
113 cylon = cylon << 1;
114 } else {
115 if (alive_led != 0) alive_led = alive_led / 2;
116 cylon = cylon >> 1;
117 }
118 if (alive_led < 2) {
119 alive_led = 2;
120 LED_UP = true;
121 } else {
122 if (alive_led > 128) {
123 alive_led = 128;
124 LED_UP = false;
125 }
126 }
127 j = 0;
128 }
129}
