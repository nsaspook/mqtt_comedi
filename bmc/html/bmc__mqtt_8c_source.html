975 */
976char * validate_bmc_text(const char * text, bool * valid)
977{
978 char tmp_test_ptr[SYSLOG_SIZ], *tmp_p = (char *) text;
979 uint32_t len = 0, starts = 0, checkmark = 0;
980 bool end_data = false;
981 char *jtoken;
982
983 validate_failure = 0;
984 strncpy(tmp_test_ptr, text, SYSLOG_SIZ - 1);
985 valid[0] = true;
986 if (tmp_test_ptr[0] == '^') {
987 starts++;
988 tmp_p = (char *) &text[0]; // return pointer to start of possible data
989 if ((len = strlen(tmp_test_ptr)) >= VALIDATE_LEN) {
990 for (int i = 1; i <= len; i++) {

992 starts++;
993 end_data = false;
